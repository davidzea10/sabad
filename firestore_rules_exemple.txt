// Copier ces règles dans Firebase Console > Firestore > Règles
// et cliquer sur "Publier" pour autoriser la collection users.

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    match /biens/{bienId} {
      allow read: if request.auth != null;
      allow create: if request.auth != null
        && request.resource.data.proprietaireId == request.auth.uid;
      allow update, delete: if request.auth != null
        && resource.data.proprietaireId == request.auth.uid;
    }

    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
  }
}
